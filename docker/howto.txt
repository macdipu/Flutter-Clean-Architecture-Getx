# how to build docker image and use for this Flutter project

# build complete-flutter image (default target)
docker buildx build --target complete-flutter -t customer-flutter-build .

# run interactive shell in container (from project root)
docker run --rm -it -v "${PWD}:/project" -v "$HOME/.dockercache/gradle:/root/.gradle" -v "$HOME/.dockercache/pub:/root/.pub-cache" customer-flutter-build bash -l

# flutter pub get
docker run --rm -v "${PWD}:/project" -v "$HOME/.dockercache/pub:/root/.pub-cache" customer-flutter-build bash -c 'cd /project && flutter pub get'

# flutter analyze
docker run --rm -v "${PWD}:/project" -v "$HOME/.dockercache/pub:/root/.pub-cache" customer-flutter-build bash -c 'cd /project && flutter analyze'

# flutter test
docker run --rm -v "${PWD}:/project" -v "$HOME/.dockercache/pub:/root/.pub-cache" customer-flutter-build bash -c 'cd /project && flutter test'

# build APK (release)
docker run --rm -v "${PWD}:/project" -v "$HOME/.dockercache/gradle:/root/.gradle" -v "$HOME/.dockercache/pub:/root/.pub-cache" customer-flutter-build bash -c 'cd /project && flutter build apk --release'

# build AAB (Play Store)
docker run --rm -v "${PWD}:/project" -v "$HOME/.dockercache/gradle:/root/.gradle" -v "$HOME/.dockercache/pub:/root/.pub-cache" customer-flutter-build bash -c 'cd /project && flutter build appbundle --release'

# create pub and gradle cache dirs (run once)
mkdir -p "$HOME/.dockercache/gradle" "$HOME/.dockercache/pub"
